//Global Constants Only, No Global Variables

const int COLMAX=80;//Only 20 required, and 1 for null terminator

//Function Prototypes Here
int  read(char [][COLMAX],int &);//Outputs row and columns detected from input
void sort(char [][COLMAX],int,int,const char[],const char[]);//Sort by row using strcmp();

//Program Execution Begins Here
int main(int argc, char** argv) {
    //Declare all Variables Here
    const int ROW=30;             //Only 20 required
    char array[ROW][COLMAX];      //Bigger than necessary 
    int colIn,colDet,rowIn,rowDet;//Row, Col input and detected
    char replace[COLMAX],with[COLMAX];//New sort order
    
    //Input the new sort order then sort

    cin>>colIn;
    
    //Now read in the array of characters and determine it's size
    rowDet=rowIn;
    cout<<"Now input the array."<<endl;
    colDet=read(array,rowDet);
    
    //Compare the size input vs. size detected and sort if same

}
int read(char arr[][COLMAX], int &rows) {
    int maxLen = 0;
    int count = 0; 
    int strSize;
    for(int i=0;i<COLMAX;i++){
        if(cin >> arr[i]){
            count++;
            strSize=strlen(arr[i]);
            if(strSize>maxLen)maxLen=strSize;
        }
    }
    rows = count; // Update rows to actual number of strings read
    return maxLen; // Return max string length
}
void sort(char arr[][COLMAX],int rows,int cols,const char rpl[],const char wth[]){//Sort by row
    char temp[COLMAX];
    for(int i=0;i<rows;i++){
        for(int j=rows-1;j>i;j--){
            if(strcmp(arr[j],arr[j-1],rpl,wth)<0){
                strcpy(temp,arr[j-1]);
                strcpy(arr[j-1], arr[j]);
                strcpy(arr[j], temp);
            }
        }
    }
}

void print(const char arr[][COLMAX],int rows,int cols){//Print the sorted 2-D array
    for(int i=0;i<rows;i++){
        cout<<arr[i]<<endl;

}
int strcmp(char arr1[],char arr2[],const char rpl[],const char wth[]){
    int size1=strlen(arr1),
        size2=strlen(arr2);
    int size3=strlen(rpl);
    int ch1[size1],
        ch2[size2];
    int  asciiDec[128];
    char asciiChar[128];
    for (int i=0;i<128;i++){
        asciiDec[i]=i;
        asciiChar[i]=static_cast<unsigned char>(i);
    }
    for(int i=0;i<128;i++){
        for(int j=0;j<size3;j++){
            if(asciiChar[i]==rpl[j]){
                asciiChar[i]=wth[j];
            }
            else if(asciiChar[i]==wth[j]){
                asciiChar[i]=rpl[j];
            }
        }
    }
    for(int i=0;i<size1;i++){
        for(int j=0;j<128;j++){
            if(arr1[i]==asciiChar[j]){
                 ch1[i]=asciiDec[j];
            }
        }
    }
    for(int i=0;i<size2;i++){
        for(int j=0;j<128;j++){
            if(arr2[i]==asciiChar[j]) ch2[i]=asciiDec[j];
        }
    }
    for(int i=0;i<size1;i++){
        if(ch1[i]>ch2[i])return 1;
        else if (ch1[i]<ch2[i])return -1;
    }
    return 0;
}
