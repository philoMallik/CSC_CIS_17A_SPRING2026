int  srch1(const char arr[],const char pat[],int pos){
    int i=0;
    int sizeA=strlen(arr);
    while(pat[i]==arr[pos]&&pos<sizeA){
        pos++;
        i++;
    }
    return pos;
}
void srchAll(const char arr[],const char pat[],int match[]){
    int count=0;
    int sizeA=strlen(arr);
    int sizeP=strlen(pat);
    int end;
    for(int i=0;i<sizeA;i++){
        end=srch1(arr,pat,i);
        if((end-i)==sizeP){
            match[count]=i;
            count++;
        }
    }
    match[count]=-1;
}
