bool  inRange(const char input[],unsigned short &unshort){
    int temp=0;
    bool end=false;
    //cout<<"step1"<<endl;
    //cout<<input<<endl;
    for(int i=0;i<5;i++){
        if(input[i]=='\0'){
            i=5;
            end=true;
        }
        else if(input[i]<48||input[i]>57) return false;
        else temp=temp*10+(input[i]-48);
    }
    if(input[5]!='\0'&&end==false) return false;
    //cout<<temp<<endl;
    unshort=static_cast<unsigned short>(temp);
    return true;
}
bool  reverse(unsigned short unshort,signed short &snshort){
    unsigned short temp=0;
    unsigned short input=unshort;
    //cout<<"step2"<<endl;
    temp=(temp*10)+((input/1    )%10);
    temp=(temp*10)+((input/10   )%10);
    temp=(temp*10)+((input/100  )%10);
    temp=(temp*10)+((input/1000 )%10);
    temp=(temp*10)+((input/10000)%10);
    //cout<<temp<<endl;
    if(temp>32767) return false;
    snshort=static_cast<signed short>(temp);
    return true;
}
short subtrct(signed short num,int val){
    //cout<<"step3"<<endl;
    if(num<=val) return num;
    else return num-val;
}
